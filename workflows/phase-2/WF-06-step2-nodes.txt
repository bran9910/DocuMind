NŒUD 1 — "Extract Admin Command"  (type: Code)
────────────────────────────────────────────────────────────────────────────────
const msg = $input.first().json.message;
const text = (msg.text || '').trim();
const chatId = msg.chat.id;
const telegramId = msg.from.id;

// Exemples : /admin stats | /admin credits 123456789 50 | /admin ban 123456789
const parts = text.split(/\s+/);
const subCommand = (parts[1] || '').toLowerCase();  // stats, credits, ban, unban, list_users, analytics
const arg1 = parts[2] || '';  // telegram_id ou @username
const arg2 = parts[3] || '';  // valeur pour credits

return [{
  json: {
    chat_id: chatId,
    telegram_id: telegramId,
    sub_command: subCommand,
    arg1: arg1.replace('@', ''),
    arg2: arg2,
    raw_text: text
  }
}];

NŒUD 10b — "Format User List"  (type: Code)
────────────────────────────────────────────────────────────────────────────────
const rows = $input.all();
const chat_id = $('Extract Admin Command').first().json.chat_id;

let lines = rows.map(r => {
  const u = r.json;
  const status = u.is_active ? '✅' : '🚫';
  const admin  = u.is_admin  ? ' 👑' : '';
  const name   = u.first_name || u.telegram_username || 'Inconnu';
  return `${status}${admin} *${name}* | ID: \`${u.telegram_id}\` | 💳${u.credits_remaining} | ${u.last_active}`;
});

return [{
  json: {
    chat_id,
    text: `👥 *LISTE UTILISATEURS (${rows.length})*\n\n` + lines.join('\n')
  }
}];

NŒUD 11a — "Query Analytics"  (type: PostgreSQL → Execute Query)
────────────────────────────────────────────────────────────────────────────────
Connecté depuis Switch output 5 (analytics)

Query :
SELECT
  TO_CHAR(created_at::date, 'DD/MM') AS jour,
  COUNT(*)                            AS sessions,
  COUNT(*) FILTER (WHERE status = 'completed')    AS ok,
  COUNT(*) FILTER (WHERE status = 'error')        AS errors,
  COALESCE(SUM(llm_tokens_used), 0)               AS tokens
FROM sessions
WHERE created_at >= NOW() - INTERVAL '7 days'
GROUP BY created_at::date
ORDER BY created_at::date DESC

(pas de queryParameters)
