const session    = $input.first().json;

NODE 12 : "Parse Custom Response"
--------------------------------------------------------------------
// Gère les deux formats de sortie du Basic LLM Chain selon version n8n
const buildInput   = $('Build Custom Prompt').first().json;
const chainOutput  = $input.first().json;

const rawOutput = (chainOutput.response && chainOutput.response.text)
               || chainOutput.text
               || chainOutput.output
               || (typeof chainOutput.response === 'string' ? chainOutput.response : '')
               || '';

let parsed = null;
let parseError = null;

// Tentative 1 : JSON direct
try {
  parsed = JSON.parse(rawOutput.trim());
} catch(e1) {
  // Tentative 2 : extraire JSON depuis texte markdown
  const jsonMatch = rawOutput.match(/\{[\s\S]*\}/);
  if (jsonMatch) {
    try {
      parsed = JSON.parse(jsonMatch[0]);
    } catch(e2) {
      parseError = e2.message;
    }
  }
}

// Fallback : structure vide si parsing échoue
if (!parsed) {
  parsed = {
    resume_executif:     rawOutput.substring(0, 500) || 'Analyse générée',
    executive_summary:   rawOutput.substring(0, 500) || 'Analysis generated',
    analyse_ciblee:      { points_cles: [], tendances_detectees: [], anomalies: [] },
    focused_analysis:    { key_findings: [], patterns_detected: [], anomalies: [] },
    insights:            [],
    recommandations:     [],
    recommendations:     [],
    adaptation_audience: '',
    audience_adaptation: '',
    score_qualite_donnees: '?',
    data_quality_score:  '?',
    limites:             parseError || 'Erreur de parsing JSON',
    limitations:         parseError || 'JSON parsing error'
  };
}

return [{
  json: {
    parsed_response: parsed,
    raw_output:      rawOutput,
    session_id:      buildInput.session_id,
    chat_id:         buildInput.chat_id,
    language:        buildInput.language,
    file_name:       buildInput.file_name,
    custom_answers:  buildInput.custom_answers,
    response_time:   Date.now()
  }
}];
